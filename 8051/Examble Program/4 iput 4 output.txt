#include <reg51.h>
#include "Delay.h"
sbit Led1  = P2^2;
sbit Led2 = P2^3;
sbit Led3 = P2^4;
sbit Led4 = P2^5;
sbit Switch1 = P1^2;
sbit Switch2 = P1^3;
sbit Switch3 = P1^4;
sbit Switch4 = P1^5;

 
 void main()
 {
	 Switch1=Switch2=Switch3=Switch4=1;
	 Switch1=Switch2=Switch3=Switch4=0;
	 Led1=Led2=Led3=Led4=0;
	 
	 while(1)
	 {
		 if (Switch1 ==1)
		 {
			 Led1 =1;
		   
		 }
		 else if (Switch2 ==1)
		 {
			 Led2 =1;
		 
	 }
		 else if (Switch3 ==1)
		 {
			 Led3 =1;
		 
	 }
		 else if (Switch4 ==1)
		 {
			 Led4 =1;
		 
	 }
		 else
		 {
			 Led1=Led2=Led3=Led4=0;
		 }
		 
 }
	 }
